魔术方法：迭代器

迭代说人话就是循环
iter():iter()
next():next()

class fibs:
def init(self):
self.a = 0
self.b = 1
def iter(self):
return self
def next(self):
self.a,self.b = self.b,self.a+self.b
return self.a

f = fibs()
for each in f:
if each < 50:
print(each)
else:
break

class fibs:
def init(self,n=50):
self.a = 0
self.b = 1
self.n = n
def iter(self):
return self
def next(self):
self.a,self.b = self.b,self.a+self.b
if self.a > self.n:
raise StopIteration
return self.a

f = fibs()
for each in f:
print(each)
