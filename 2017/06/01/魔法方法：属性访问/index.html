class Rectangle:
def init(self,width=0,height=0):
self.width = width
self.height = height

def __setattr__(self,name,value):
    print('using setattr')
    if name == 'square':
        self.width = value
        self.height = value
    else:
        super().__setattr__(name,value)
        #self.__dict__[name] = value

def getArea(self):
